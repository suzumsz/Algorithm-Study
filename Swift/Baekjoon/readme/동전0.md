### Sol
파이썬으로 풀어본 후 푼 코드이다.   
다음 동전들이 이전 동전의 각 배수이므로, 그리디로 풀 수 있었다.
```Swift
// n,k 입력받기
let input = readLine()!.split(separator : " " ).map{Int(String($0))!}
let n = input[0]
var k = input[1]

// 동전 1원부터 50000원까지 입력받기 
var list = [Int]()
for _ in 0..<n {
    list.append(Int(readLine()!)!)
}

// 가장 큰 동전부터 몫을 카운트 해주고, 나머지 동전을 내림차순으로 반복
var count = 0
for i in (0..<n).reversed() {
    count += (k/list[i])
    k %= list[i]
}

print(count)
```
