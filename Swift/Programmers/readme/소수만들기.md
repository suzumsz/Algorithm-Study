### Sol
삼중 for문을 사용해 세개의 값을 비교하면서 더해 sums에 넣었다.  
그리고 filter와 isPrime을 이용하여 소수가 맞는 수의 갯수를 센 뒤 반환하여 풀 수 있었다.  
블로그를 참고하여 풀었다ㅜㅜ 더 공부하자ㅜ
```Swift
import Foundation

func solution(_ nums:[Int]) -> Int {
    var sums = [Int]()

    for i in 0..<nums.count-2 {
        for j in i+1..<nums.count {
            for k in j+1..<nums.count {
                sums.append(nums[i] + nums[j] + nums[k])
            }
        }
    }
    return  sums.filter{isPrime($0)}.count
}

func isPrime(_ n:Int) -> Bool {
    for i in 2..<n {
        if n%i == 0 {
            return false
        }
    }
    return true
}
```
